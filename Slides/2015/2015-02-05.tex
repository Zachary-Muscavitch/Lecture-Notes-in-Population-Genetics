\myslide{
\begin{eqnarray*}
\mbox{P}(i|k) &=& \frac{\mbox{P}(x_i|\gamma_k)}{\sum_k
  \mbox{P}(x_i|\gamma_k)} \\
x_i &=& \mbox{genotype of individual $i$} \\
\gamma_k &=& \mbox{genotype frequencies in population $k$}
\end{eqnarray*}
For example,
\begin{eqnarray*}
\mbox{P}((1,2,2,1,3)|(p_1, p_2, p_3, p_4, p_5)) = (p_1^2)(2p_2q_2)(2p_3q_3)(p_4^2)(q_5^2)
\end{eqnarray*}
}
